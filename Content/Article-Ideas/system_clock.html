Using the System Clock

To get the time from the system clock, include the "time.h" C library.

#include <ctime>

The function time(time_t *time); returns the number of seconds that have elapsed since January 1, 1970, according to your systems clock. The return type is time_t.

time_t now = time( 0 );

Here is a sample program: 

// example_time_program.cpp
	
#include <ctime>
#include <iostream>
using namespace std;

int main()
{
	time_t seconds = time( 0 ); // seconds since January 1, 1970
	time_t minutes = seconds / 60;
	time_t hours = minutes / 60;
	time_t days = hours / 24;
	time_t years = 0; 
	time_t days_not_counted = days;
	int leap_year = 2;
	while( days_not_counted > 365 )
	{
		if( leap_year % 4 == 0 )
		{
			days_not_counted -= 366;
		}
		else //( not a leap year )
		{
			days_not_counted -= 365;
		}
		years++;
		leap_year++;
	}
		
	cout << "Year: " << years + 1970 << endl;
	cout << "Days: " << days_not_counted << endl;
	const GMT_TO_EST = -4; // if it is 8:00 p.m. in England, it is 4:00 p.m. in New York
	cout << "Hour: " << hours % 24 + GMT_TO_EST << endl;
}
