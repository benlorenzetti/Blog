Getting GNU Scientific Library

$ apt-cache search gsl
$ sudo apt-get install libgsl0-dev libgsl-ref-html -y
$ firefox /usr/share/doc/gsl-ref-html/index.html

Getting Gnuplot

$ sudo apt-get install gnuplot gnuplot-x11 gnuplot-doc -y
$ firefox /usr/share/doc/gnuplot-doc/htmldocs/index.html

Getting Nicolas Devillard's C Interface for Gnuplot

ndevilla.free.fr/gnuplot

wget http://ndevilla.free.fr/gnuplot/gnuplot_i-2.11.tar.gz
tar xvf gnuplot_i-2.11.tar.gz
rm gnuplot_i-2.11.tar.gz
cd gnuplot_i/src/
cp gnuplot_i.h ../../
cp gnuplot_i.c ../../
cd ../../
rm -rf gnuplot_i
gcc -O3 -c gnuplot_i.c
